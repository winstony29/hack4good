```jsx
  const variants = {

    info: 'bg-minds-teal/10 text-minds-teal border-minds-teal/20',
    success: 'bg-green-50 text-green-700 border-green-200',
    warning: 'bg-minds-amber/10 text-amber-700 border-minds-amber/30',
    danger: 'bg-red-50 text-red-700 border-red-200',
    primary: 'bg-minds-coral/10 text-minds-coral border-minds-coral/20',
    // Accessibility-specific badges
    accessible: 'bg-minds-teal/10 text-minds-teal border-minds-teal/20',
    limited: 'bg-gray-100 text-minds-gray border-gray-200',
  }

  return (
    <span
      className={`
        inline-flex items-center gap-1.5
        px-3 py-1.5
        text-sm font-medium
        rounded-full
        border
        ${variants[variant]}
        ${className}
      `}
      {...props}
    >
      {children}
    </span>
  )
}
```

Update ActivityCard with the inclusive design:

```jsx
import { Calendar, Clock, MapPin, Users, DollarSign, Wheelchair, ArrowRight } from 'lucide-react'
import { motion } from 'framer-motion'
import Badge from '../shared/Badge'
import TTSButton from '../accessibility/TTSButton'
import { formatDate, formatTime } from '../../utils/dateUtils'
import { getActivityTitle, getActivityDescription } from '../../utils/activityTranslations'
import { useAccessibility } from '../../contexts/AccessibilityContext'

export default function ActivityCard({ activity, showCapacity = true, showTTS = false, onClick }) {
  const { language, reduceMotion } = useAccessibility()
  const availableSpots = activity.max_capacity - activity.current_participants
  const isFull = availableSpots <= 0

  // Get translated content
  const title = getActivityTitle(activity, language)
  const description = getActivityDescription(activity, language)

  // Program type colors
  const programColors = {
    sports: 'bg-blue-50 text-blue-700',
    arts: 'bg-purple-50 text-purple-700',
    music: 'bg-pink-50 text-pink-700',
    social: 'bg-minds-teal/10 text-minds-teal',
    educational: 'bg-minds-amber/10 text-amber-700',
    wellness: 'bg-green-50 text-green-700',
    default: 'bg-minds-coral/10 text-minds-coral',
  }

  const programColor = programColors[activity.program_type?.toLowerCase()] || programColors.default

  return (
    <motion.div
      whileHover={!reduceMotion ? { y: -4 } : undefined}
      transition={{ type: 'spring', stiffness: 300, damping: 25 }}
      onClick={onClick}
      className="
        bg-white rounded-2xl
        shadow-soft hover:shadow-card-hover
        border border-minds-border
        cursor-pointer
        overflow-hidden
        transition-shadow duration-200
      "
      style={{
        backgroundColor: 'var(--a11y-bg)',
        borderColor: 'var(--a11y-border)',
      }}
      role="article"
      tabIndex={0}
      onKeyDown={(e) => {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault()
          onClick?.()
        }
      }}
      aria-label={`${title} on ${formatDate(activity.date)}`}
    >
      {/* Color accent bar at top */}
      <div className={`h-2 ${programColor.split(' ')[0]}`} />

      <div className="p-5">
        {/* Header: Title + Program Badge */}
        <div className="flex justify-between items-start gap-3 mb-3">
          <h3 className="text-xl font-bold text-minds-charcoal leading-tight flex-1">
            {title}
          </h3>
          {activity.program_type && (
            <span className={`px-3 py-1 rounded-full text-sm font-medium ${programColor}`}>
              {activity.program_type}
            </span>
          )}
        </div>

        {/* Description */}
        <p className="text-minds-gray mb-4 line-clamp-2 leading-relaxed">
          {description}
        </p>

        {/* Accessibility & Payment Badges */}
        <div className="flex flex-wrap gap-2 mb-4">
          {activity.wheelchair_accessible && (
            <Badge variant="accessible" className="flex items-center gap-1.5">
              <Wheelchair className="w-4 h-4" />
              <span>Wheelchair Accessible</span>
            </Badge>
          )}
          {activity.wheelchair_accessible === false && (
            <Badge variant="limited" className="flex items-center gap-1.5">
              <Wheelchair className="w-4 h-4" />
              <span>Limited Access</span>
            </Badge>
          )}
          {activity.payment_required && (
            <Badge variant="warning" className="flex items-center gap-1.5">
              <DollarSign className="w-4 h-4" />
              <span>Fee Required</span>
            </Badge>
          )}
        </div>

        {/* Event Details */}
        <div className="space-y-2.5 text-minds-charcoal">
          <div className="flex items-center gap-3">
            <div className="w-8 h-8 bg-minds-coral/10 rounded-lg flex items-center justify-center flex-shrink-0">
              <Calendar className="w-4 h-4 text-minds-coral" />
            </div>
            <span className="font-medium">{formatDate(activity.date)}</span>
          </div>
          <div className="flex items-center gap-3">
            <div className="w-8 h-8 bg-minds-teal/10 rounded-lg flex items-center justify-center flex-shrink-0">
              <Clock className="w-4 h-4 text-minds-teal" />
            </div>
            <span>{formatTime(activity.start_time)} - {formatTime(activity.end_time)}</span>
          </div>
          <div className="flex items-center gap-3">
            <div className="w-8 h-8 bg-minds-amber/10 rounded-lg flex items-center justify-center flex-shrink-0">
              <MapPin className="w-4 h-4 text-minds-amber" />
            </div>
            <span className="line-clamp-1">{activity.location}</span>
          </div>
        </div>

        {/* Capacity & CTA */}
        {showCapacity && (
          <div className="mt-5 pt-4 border-t border-minds-border flex items-center justify-between">
            <div className="flex items-center gap-2">
              <Users className="w-5 h-5 text-minds-gray" />
              <span className="text-minds-gray">
                {activity.current_participants}/{activity.max_capacity}
              </span>
              <Badge variant={isFull ? 'danger' : availableSpots <= 5 ? 'warning' : 'success'}>
                {isFull ? 'Full' : `${availableSpots} spots`}
              </Badge>
            </div>
            <span className="text-minds-coral font-semibold flex items-center gap-1">
              Learn More
              <ArrowRight className="w-4 h-4" />
            </span>
          </div>
        )}

        {/* TTS Button */}
        {showTTS && (
          <div className="mt-4 pt-4 border-t border-minds-border">
            <TTSButton text={`${title}. ${description}`} />
          </div>
        )}
      </div>
    </motion.div>
  )
}
```

Key changes:
- Color accent bar at top based on program type
- Icons in colored rounded squares
- Better visual hierarchy with bold title
- Keyboard accessible (role, tabIndex, onKeyDown)
- "Learn More" CTA prompt
- Warm color palette throughout
